Deploy the Apps Script below in your sheet to enable write-back tagging.
Apps Script setup — paste this in your sheet's script editor (Extensions → Apps Script), then deploy as a web app (Execute as: Me, Access: Anyone):

function doPost(e){
  var d=JSON.parse(e.postData.contents);
  var s=SpreadsheetApp.getActiveSpreadsheet()
    .getActiveSheet();
  s.getRange(d.row,d.col).setValue(d.val);
  return ContentService
    .createTextOutput('ok');
}
